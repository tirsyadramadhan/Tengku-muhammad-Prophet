PO Table:
CREATE TABLE IF NOT EXISTS `tbl_po` (
  `po_id` bigint NOT NULL AUTO_INCREMENT,
  `customer_id` bigint NOT NULL,
  `no_po` varchar(100) CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci NOT NULL,
  `nama_barang` varchar(255) CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci NOT NULL,
  `tgl_po` date NOT NULL,
  `periode_po` varchar(7) CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci DEFAULT NULL COMMENT 'Format: YYYY-MM',
  `qty` bigint NOT NULL DEFAULT '0',
  `harga` decimal(15,2) NOT NULL DEFAULT '0.00',
  `total` decimal(15,2) NOT NULL DEFAULT '0.00',
  `modal_awal` decimal(15,2) NOT NULL DEFAULT '0.00',
  `margin` decimal(15,2) NOT NULL DEFAULT '0.00',
  `margin_unit` decimal(15,2) NOT NULL DEFAULT '0.00',
  `status` tinyint NOT NULL DEFAULT '0' COMMENT '0: Open, 1: Delivered, 2: Invoiced, 3: Closed',
  `input_by` bigint DEFAULT NULL,
  `input_date` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `edit_by` bigint DEFAULT NULL,
  `edit_date` datetime DEFAULT NULL ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (`po_id`),
  UNIQUE KEY `uk_no_po` (`no_po`),
  KEY `idx_customer` (`customer_id`),
  KEY `idx_status` (`status`),
  KEY `idx_tgl_po` (`tgl_po`),
  KEY `fk_po_input_user` (`input_by`),
  CONSTRAINT `fk_po_customer` FOREIGN KEY (`customer_id`) REFERENCES `tbl_customer` (`id_cust`) ON DELETE RESTRICT,
  CONSTRAINT `fk_po_input_user` FOREIGN KEY (`input_by`) REFERENCES `tbl_user` (`user_id`) ON DELETE SET NULL
)

Delivery Table:
CREATE TABLE IF NOT EXISTS `tbl_delivery` (
  `delivery_id` bigint NOT NULL AUTO_INCREMENT,
  `po_id` bigint NOT NULL,
  `delivery_no` varchar(50) CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci NOT NULL,
  `qty_delivered` decimal(15,2) NOT NULL DEFAULT '0.00',
  `delivery_time_estimation` datetime DEFAULT NULL,
  `delivered_at` datetime DEFAULT NULL,
  `delivered_status` tinyint NOT NULL DEFAULT '0' COMMENT '0: Pending, 1: Shipped, 2: Delivered',
  `invoiced_status` tinyint NOT NULL DEFAULT '0' COMMENT '0: Uninvoiced, 1: Invoiced',
  `input_by` bigint DEFAULT NULL,
  `input_date` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `edit_by` bigint DEFAULT NULL,
  `edit_date` datetime DEFAULT NULL ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (`delivery_id`),
  UNIQUE KEY `uk_delivery_no` (`delivery_no`),
  KEY `idx_delivery_po` (`po_id`),
  KEY `fk_delivery_input` (`input_by`),
  CONSTRAINT `fk_delivery_input` FOREIGN KEY (`input_by`) REFERENCES `tbl_user` (`user_id`) ON DELETE SET NULL,
  CONSTRAINT `fk_delivery_po` FOREIGN KEY (`po_id`) REFERENCES `tbl_po` (`po_id`) ON DELETE RESTRICT ON UPDATE CASCADE
)

Invoice Table:
CREATE TABLE IF NOT EXISTS `tbl_invoice` (
  `invoice_id` bigint NOT NULL AUTO_INCREMENT,
  `delivery_id` bigint NOT NULL,
  `nomor_invoice` varchar(255) CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci DEFAULT NULL,
  `tgl_invoice` datetime DEFAULT NULL,
  `due_date` datetime DEFAULT NULL,
  `status_invoice` tinyint DEFAULT '0' COMMENT '0: Unpaid, 1: Paid, 2: Cancelled',
  `input_by` bigint DEFAULT NULL,
  `input_date` datetime DEFAULT CURRENT_TIMESTAMP,
  `edit_by` bigint DEFAULT NULL,
  `edit_date` datetime DEFAULT NULL ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (`invoice_id`),
  UNIQUE KEY `uk_nomor_invoice` (`nomor_invoice`),
  KEY `idx_invoice_delivery` (`delivery_id`),
  CONSTRAINT `fk_invoice_delivery` FOREIGN KEY (`delivery_id`) REFERENCES `tbl_delivery` (`delivery_id`) ON DELETE RESTRICT
)

PO Statuses =
0: Incoming – The PO is arriving in the system or being drafted.
1: Open – The PO is approved and active. We are waiting for the vendor to ship.

2: Partially Delivered – The truck arrived with some items. No invoices have been entered yet.
process: first check if a PO has a delivery if yes then check how many delivery is linked to that po then get the qty_delivered column from each delivery then sum it up 
then if the summed value is less than the qty of that PO then the PO status is set to 2 (Partially Delivered)

3: Fully Delivered – Everything has arrived at the warehouse. We are just waiting for the paperwork (invoice) to catch up.
process: first check if a PO has a delivery if yes then check how many delivery is linked to that po then get the qty_delivered column from each delivery then sum it up 
then if the summed value is the same as the qty of that PO then the PO status is set to 3 (Fully Delivered)

4: Partially Delivered & Partially Invoiced – The standard "ongoing" status. We received some goods and entered the invoice for those specific goods.
process: first check if a PO has a delivery if yes then check how many delivery is linked to that po then get the qty_delivered column from each delivery then sum it up 
then if the summed value is less than the qty of that PO then Partially Delivered is already here then check if there's any delivery with invoiced_status = 1 if yes then 
count how many delivery has the invoiced_status = 1 if not all of the delivery has the invoiced_status = 1 then it's Partially Invoiced so now with both 
Partially Delivered & Partially Invoiced set the status of that PO to 4 (Partially Delivered & Partially Invoiced)

5: Fully Delivered & Partially Invoiced – We have all the goods, but the vendor has only billed us for a portion of them so far.
process: first check if a PO has a delivery if yes then check how many delivery is linked to that po then get the qty_delivered column from each delivery then sum it up 
then if the summed value is the same as the qty of that PO then it's Fully Delivered then check if there's any delivery with invoiced_status = 1 if yes then count how 
many delivery has the invoiced_status = 1 if not all of the delivery has the invoiced_status = 1 then it's Partially Invoiced so now with both 
Fully Delivered & Partially Invoiced set the status of that PO to 5 (Fully Delivered & Partially Invoiced)

6: Partially Delivered & Fully Invoiced – Rare but possible. We received a partial shipment, but the vendor sent the bill for the entire order (and we entered it).
process: first check if a PO has a delivery if yes then check how many delivery is linked to that po then get the qty_delivered column from each delivery then sum it up 
then if the summed value is less than the qty of that PO then Partially Delivered is already here then check if there's any delivery with invoiced_status = 1 if yes 
then count how many delivery has the invoiced_status = 1 if all of the delivery has the invoiced_status = 1 then it's Fully Invoiced so now with both 
Partially Delivered & Fully Invoiced set the status of that PO to 6 (Partially Delivered & Fully Invoiced)

7: FULLY DELIVERED AND FULLY INVOICED – The loop is closed. All goods are received, and the full amount is invoiced (Fully Delivered & Fully Invoiced).

8: Closed
PROCESS: MUST BE FULLY DELIVERED AND FULLY INVOICED but WAIT were all invoices paid yet? if not keep the status to FULLY DELIVERED AND FULLY INVOICED until
every invoice has the status_invoice = 1
